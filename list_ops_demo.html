✏️ 2-2. 列表操作示例演示

list_methods.py
# 定义列表 fruits
fruits = ["apple", "banana", "cherry"]
print("1. 初始列表:", fruits)

# 2. append()：末尾添加
fruits.append("orange")
print("2. append后:", fruits)

# 3. extend()：批量添加
fruits.extend(["pear", "peach"])
print("3. extend后:", fruits)

# 4. insert()：插入
fruits.insert(1, "grape")
print("4. insert后:", fruits)

# 5. remove()：删除指定元素
fruits.remove("banana")
print("5. remove后:", fruits)

# 6. pop()：弹出
popped = fruits.pop()
print("6. 弹出的:", popped)
print("   pop后:", fruits)

# 7. sort()：排序
fruits.sort()
print("7. sort后:", fruits)
